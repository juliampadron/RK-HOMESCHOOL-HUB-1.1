# Student Dashboard Pages

Comprehensive dashboard interface for students and parents to view enrollments, games, progress, and portfolio.

## Main Dashboard Page

```tsx
'use client';

import { useEffect, useState } from 'react';
import { createClient } from '@supabase/supabase-js';
import Link from 'next/link';

interface DashboardProps {
  studentId: string;
}

export default function StudentDashboard({ studentId }: DashboardProps) {
  const [student, setStudent] = useState<any>(null);
  const [enrollments, setEnrollments] = useState<any[]>([]);
  const [games, setGames] = useState<any[]>([]);
  const [recentProgress, setRecentProgress] = useState<any[]>([]);
  const [portfolioCount, setPortfolioCount] = useState(0);
  const [loading, setLoading] = useState(true);

  const supabase = createClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!
  );

  useEffect(() => {
    async function loadDashboardData() {
      try {
        // Load student info
        const { data: studentData } = await supabase
          .from('students')
          .select('*, profiles(*), families(*)')
          .eq('id', studentId)
          .single();

        setStudent(studentData);

        // Load active enrollments
        const { data: enrollmentData } = await supabase
          .from('enrollments')
          .select('*, classes(*, profiles(username, full_name))')
          .eq('student_id', studentId)
          .eq('status', 'active')
          .order('enrollment_date', { ascending: false });

        setEnrollments(enrollmentData || []);

        // Load accessible games (public + enrolled)
        const { data: gameData } = await supabase
          .from('games')
          .select('*')
          .or(`is_public.eq.true,id.in.(${await getEnrolledGameIds()})`);

        setGames(gameData || []);

        // Load recent progress
        const { data: progressData } = await supabase
          .from('student_progress')
          .select('*, games(title, slug, subject_area)')
          .eq('student_id', studentId)
          .order('last_played_at', { ascending: false })
          .limit(5);

        setRecentProgress(progressData || []);

        // Count portfolio items
        const { count } = await supabase
          .from('portfolio_items')
          .select('*', { count: 'exact', head: true })
          .eq('student_id', studentId);

        setPortfolioCount(count || 0);

      } catch (error) {
        console.error('Error loading dashboard:', error);
      } finally {
        setLoading(false);
      }
    }

    async function getEnrolledGameIds() {
      const { data } = await supabase
        .from('game_enrollments')
        .select('game_id')
        .eq('student_id', studentId);

      return data?.map(e => e.game_id).join(',') || '';
    }

    loadDashboardData();
  }, [studentId]);

  if (loading) {
    return <div className="loading">Loading dashboard...</div>;
  }

  return (
    <div className="dashboard-container">
      <header className="dashboard-header">
        <div>
          <h1>Welcome back, {student?.profiles?.full_name || 'Student'}!</h1>
          <p className="grade-level">Grade {student?.grade_level || 'N/A'}</p>
        </div>
        <div className="header-actions">
          <Link href="/classes/browse" className="btn btn-primary">
            Browse Classes
          </Link>
        </div>
      </header>

      <div className="dashboard-grid">
        {/* Stats Cards */}
        <div className="stats-row">
          <div className="stat-card">
            <div className="stat-icon">📚</div>
            <div className="stat-content">
              <div className="stat-value">{enrollments.length}</div>
              <div className="stat-label">Active Classes</div>
            </div>
          </div>

          <div className="stat-card">
            <div className="stat-icon">🎮</div>
            <div className="stat-content">
              <div className="stat-value">{games.length}</div>
              <div className="stat-label">Games Available</div>
            </div>
          </div>

          <div className="stat-card">
            <div className="stat-icon">📁</div>
            <div className="stat-content">
              <div className="stat-value">{portfolioCount}</div>
              <div className="stat-label">Portfolio Items</div>
            </div>
          </div>

          <div className="stat-card">
            <div className="stat-icon">⭐</div>
            <div className="stat-content">
              <div className="stat-value">{recentProgress.length}</div>
              <div className="stat-label">Games Played</div>
            </div>
          </div>
        </div>

        {/* Active Enrollments */}
        <section className="dashboard-section">
          <h2>My Classes</h2>
          {enrollments.length === 0 ? (
            <div className="empty-state">
              <p>You're not enrolled in any classes yet.</p>
              <Link href="/classes/browse" className="btn btn-secondary">
                Browse Classes
              </Link>
            </div>
          ) : (
            <div className="class-grid">
              {enrollments.map((enrollment) => (
                <div key={enrollment.id} className="class-card">
                  <div className="class-header">
                    <h3>{enrollment.classes.title}</h3>
                    <span className="class-subject">{enrollment.classes.subject_area}</span>
                  </div>
                  <p className="class-instructor">
                    Instructor: {enrollment.classes.profiles.full_name}
                  </p>
                  <p className="class-dates">
                    {new Date(enrollment.classes.start_date).toLocaleDateString()} -
                    {new Date(enrollment.classes.end_date).toLocaleDateString()}
                  </p>
                  <Link href={`/classes/${enrollment.classes.id}`} className="class-link">
                    View Class →
                  </Link>
                </div>
              ))}
            </div>
          )}
        </section>

        {/* Games Section */}
        <section className="dashboard-section">
          <div className="section-header">
            <h2>Learning Games</h2>
            <Link href="/games" className="view-all">View All →</Link>
          </div>
          <div className="game-grid">
            {games.slice(0, 6).map((game) => (
              <Link href={`/games/${game.slug}`} key={game.id} className="game-card">
                <div className="game-icon">🎮</div>
                <h3>{game.title}</h3>
                <p className="game-subject">{game.subject_area}</p>
                {game.is_public && <span className="free-badge">Free</span>}
              </Link>
            ))}
          </div>
        </section>

        {/* Recent Progress */}
        <section className="dashboard-section">
          <h2>Recent Activity</h2>
          {recentProgress.length === 0 ? (
            <div className="empty-state">
              <p>No recent activity. Start playing games to see your progress!</p>
            </div>
          ) : (
            <div className="progress-list">
              {recentProgress.map((progress) => (
                <div key={progress.id} className="progress-item">
                  <div className="progress-info">
                    <h4>{progress.games.title}</h4>
                    <p className="progress-date">
                      Last played: {new Date(progress.last_played_at).toLocaleDateString()}
                    </p>
                  </div>
                  <div className="progress-stats">
                    <div className="progress-stat">
                      <span className="stat-label">Score</span>
                      <span className="stat-value">{progress.score}</span>
                    </div>
                    <div className="progress-stat">
                      <span className="stat-label">Completion</span>
                      <span className="stat-value">{progress.completion_percentage}%</span>
                    </div>
                  </div>
                </div>
              ))}
            </div>
          )}
        </section>

        {/* Portfolio Section */}
        <section className="dashboard-section">
          <div className="section-header">
            <h2>Portfolio</h2>
            <Link href="/portfolio" className="view-all">View All →</Link>
          </div>
          <div className="portfolio-preview">
            <p>{portfolioCount} items in your portfolio</p>
            <Link href="/portfolio/add" className="btn btn-secondary">
              Add Portfolio Item
            </Link>
          </div>
        </section>
      </div>

        }

        }

        }

        }

        }

          background: #d94d1c;
        }

        .btn-secondary {
          background: #FBC440;

        }
```

## Portfolio Page

```tsx
'use client';

import { useEffect, useState } from 'react';
import { createClient } from '@supabase/supabase-js';

interface PortfolioPageProps {
  studentId: string;
}

export default function PortfolioPage({ studentId }: PortfolioPageProps) {
  const [items, setItems] = useState<any[]>([]);
  const [filter, setFilter] = useState<string>('all');
  const [loading, setLoading] = useState(true);

  const supabase = createClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!
  );

  useEffect(() => {
    loadPortfolioItems();
  }, [studentId, filter]);

  async function loadPortfolioItems() {
    let query = supabase
      .from('portfolio_items')
      .select('*')
      .eq('student_id', studentId)
      .order('date_completed', { ascending: false });

    if (filter !== 'all') {
      query = query.eq('subject_area', filter);
    }

    const { data, error } = await query;

    if (!error) {
      setItems(data || []);
    }

    setLoading(false);
  }

  const subjects = ['all', 'Math', 'Science', 'Language Arts', 'Social Studies', 'Arts'];

  return (
    <div className="portfolio-container">
      <header className="portfolio-header">
        <h1>My Portfolio</h1>
        <p>Collection of your learning achievements and work samples</p>
      </header>

      <div className="portfolio-filters">
        {subjects.map((subject) => (
          <button
            key={subject}
            className={`filter-btn ${filter === subject ? 'active' : ''}`}
            onClick={() => setFilter(subject)}
          >
            {subject === 'all' ? 'All Subjects' : subject}
          </button>
        ))}
      </div>

      {loading ? (
        <div className="loading">Loading portfolio...</div>
      ) : items.length === 0 ? (
        <div className="empty-state">
          <p>No portfolio items yet. Add your first work sample!</p>
        </div>
      ) : (
        <div className="portfolio-grid">
          {items.map((item) => (
            <div key={item.id} className="portfolio-item">
              <div className="item-header">
                <h3>{item.title}</h3>
                <span className="item-type">{item.item_type}</span>
              </div>
              <p className="item-description">{item.description}</p>
              <div className="item-meta">
                <span>{item.subject_area}</span>
                <span>{new Date(item.date_completed).toLocaleDateString()}</span>
              </div>
              {item.nys_standards && item.nys_standards.length > 0 && (
                <div className="standards">
                  {item.nys_standards.map((std: string, i: number) => (
                    <span key={i} className="standard-badge">{std}</span>
                  ))}
                </div>
              )}
            </div>
          ))}
        </div>
      )}

      <style jsx>{`
        .portfolio-container {
          max-width: 1200px;
          margin: 0 auto;
          padding: 24px;
        }

        .portfolio-header {
          margin-bottom: 32px;
        }

        .portfolio-header h1 {
          margin: 0 0 8px 0;
          color: #2F6B65;
        }

        .portfolio-header p {
          margin: 0;
          color: #666;
        }

        .portfolio-filters {
          display: flex;
          gap: 8px;
          margin-bottom: 24px;
          flex-wrap: wrap;
        }

        .filter-btn {
          padding: 8px 16px;
          border: 2px solid #e0e0e0;
          border-radius: 20px;
          background: white;
          font-weight: 600;
          cursor: pointer;
          transition: all 0.2s;
        }

        .filter-btn:hover {
          border-color: #2F6B65;
        }

        .filter-btn.active {
          background: #2F6B65;
          color: white;
          border-color: #2F6B65;
        }

        .portfolio-grid {
          display: grid;

          gap: 20px;
        }

        .portfolio-item {
          padding: 20px;

          box-shadow: 0 2px 8px rgba(0,0,0,0.1);
        }

        .item-header {
          display: flex;

          align-items: flex-start;
          margin-bottom: 12px;
        }

        .item-header h3 {
          margin: 0;

        }

        .item-type {

          padding: 4px 8px;
          border-radius: 4px;
          font-size: 12px;
          font-weight: 600;
        }

        .item-description {
          margin: 0 0 12px 0;
          color: #666;
          font-size: 14px;
        }

        .item-meta {
          display: flex;
```

## Features

- **Main Dashboard**: Overview of classes, games, progress, and portfolio
- **Portfolio Page**: Filterable view of student work samples
- **Real-time Stats**: Active classes, available games, portfolio count
- **Recent Activity**: Last played games with scores and completion
- **Responsive Design**: Mobile-friendly grid layouts
- **RLS Security**: All data fetched respects row-level security policies
