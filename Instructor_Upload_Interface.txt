# Instructor Upload Interface

A comprehensive interface for instructors to upload games, materials, and resources.

## Implementation

```tsx
'use client';

import { useState, useRef } from 'react';
import { createClient } from '@supabase/supabase-js';

interface InstructorUploadProps {
  instructorId: string;
  classId?: string;
}

export default function InstructorUpload({ instructorId, classId }: InstructorUploadProps) {
  const [uploading, setUploading] = useState(false);
  const [uploadType, setUploadType] = useState<'game' | 'material'>('material');
  const [formData, setFormData] = useState({
    title: '',
    description: '',
    fileType: 'document',
    subject_area: '',
    grade_levels: '',
    is_public: false,
    requires_enrollment: false,
    nys_standards: '',
  });
  const [file, setFile] = useState<File | null>(null);
  const [uploadProgress, setUploadProgress] = useState(0);
  const [message, setMessage] = useState<{ type: 'success' | 'error'; text: string } | null>(null);
  const fileInputRef = useRef<HTMLInputElement>(null);

  const supabase = createClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!
  );

  const handleFileSelect = (e: React.ChangeEvent<HTMLInputElement>) => {
    const selectedFile = e.target.files?.[0];
    if (selectedFile) {
      setFile(selectedFile);

      // Auto-detect file type
      const extension = selectedFile.name.split('.').pop()?.toLowerCase();
      let fileType = 'other';

      if (['html', 'htm'].includes(extension || '')) fileType = 'game';
      else if (['pdf'].includes(extension || '')) fileType = 'document';
      else if (['ppt', 'pptx'].includes(extension || '')) fileType = 'presentation';
      else if (['mp4', 'mov', 'avi'].includes(extension || '')) fileType = 'video';
      else if (['png', 'jpg', 'jpeg', 'gif', 'svg'].includes(extension || '')) fileType = 'image';
      else if (['doc', 'docx', 'txt'].includes(extension || '')) fileType = 'worksheet';

      setFormData(prev => ({ ...prev, fileType }));
    }
  };

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();

    if (!file) {
      setMessage({ type: 'error', text: 'Please select a file to upload' });
      return;
    }

    setUploading(true);
    setMessage(null);
    setUploadProgress(0);

    try {
      // Upload file to Supabase Storage
      const fileExt = file.name.split('.').pop();
      const fileName = `${Date.now()}_${Math.random().toString(36).substring(7)}.${fileExt}`;
      const filePath = `${uploadType}s/${instructorId}/${fileName}`;

      const { data: uploadData, error: uploadError } = await supabase.storage
        .from('instructor-uploads')
        .upload(filePath, file, {
          cacheControl: '3600',
          upsert: false,
          onUploadProgress: (progress) => {
            setUploadProgress((progress.loaded / progress.total) * 100);
          },
        });

      if (uploadError) throw uploadError;

      // Get public URL
      const { data: { publicUrl } } = supabase.storage
        .from('instructor-uploads')
        .getPublicUrl(filePath);

      // Create database record based on upload type
      if (uploadType === 'game') {
        // Insert into games table
        const { error: gameError } = await supabase.from('games').insert({
          title: formData.title,
          slug: formData.title.toLowerCase().replace(/[^a-z0-9]+/g, '-'),
          description: formData.description,
          subject_area: formData.subject_area,
          grade_levels: formData.grade_levels,
          file_path: publicUrl,
          is_public: formData.is_public,
          requires_enrollment: formData.requires_enrollment,
          class_id: classId || null,
          nys_standards: formData.nys_standards ? formData.nys_standards.split(',').map(s => s.trim()) : [],
          created_by: instructorId,
        });

        if (gameError) throw gameError;
      } else {
        // Insert into instructor_uploads table
        const { error: materialError } = await supabase.from('instructor_uploads').insert({
          instructor_id: instructorId,
          class_id: classId || null,
          title: formData.title,
          description: formData.description,
          file_type: formData.fileType,
          file_url: publicUrl,
          file_size_bytes: file.size,
          mime_type: file.type,
          is_public: formData.is_public,
        });

        if (materialError) throw materialError;
      }

      setMessage({
        type: 'success',
        text: `${uploadType === 'game' ? 'Game' : 'Material'} uploaded successfully!`
      });

      // Reset form
      setFormData({
        title: '',
        description: '',
        fileType: 'document',
        subject_area: '',
        grade_levels: '',
        is_public: false,
        requires_enrollment: false,
        nys_standards: '',
      });
      setFile(null);
      if (fileInputRef.current) fileInputRef.current.value = '';
      setUploadProgress(0);

    } catch (error: any) {
      console.error('Upload error:', error);
      setMessage({ type: 'error', text: error.message || 'Upload failed' });
    } finally {
      setUploading(false);
    }
  };

  return (
    <div className="instructor-upload-container">
      <div className="upload-header">
        <h2>Upload Content</h2>
        <p>Share games, worksheets, and materials with your students</p>
      </div>

      <div className="upload-type-selector">
        <button
          type="button"
          className={`type-btn ${uploadType === 'material' ? 'active' : ''}`}
          onClick={() => setUploadType('material')}
        >
          📄 Material/Worksheet
        </button>
        <button
          type="button"
          className={`type-btn ${uploadType === 'game' ? 'active' : ''}`}
          onClick={() => setUploadType('game')}
        >
          🎮 Interactive Game
        </button>
      </div>

      <form onSubmit={handleSubmit} className="upload-form">
        <div className="form-group">
          <label htmlFor="title">Title *</label>
          <input
            type="text"
            id="title"
            required
            value={formData.title}
            onChange={(e) => setFormData({ ...formData, title: e.target.value })}
            placeholder="Enter a clear, descriptive title"
          />
        </div>

        <div className="form-group">
          <label htmlFor="description">Description</label>
          <textarea
            id="description"
            rows={4}
            value={formData.description}
            onChange={(e) => setFormData({ ...formData, description: e.target.value })}
            placeholder="Describe the content and learning objectives"
          />
        </div>

        {uploadType === 'game' && (
          <>
            <div className="form-row">
              <div className="form-group">
                <label htmlFor="subject_area">Subject Area</label>
                <select
                  id="subject_area"
                  value={formData.subject_area}
                  onChange={(e) => setFormData({ ...formData, subject_area: e.target.value })}
                >
                  <option value="">Select subject</option>
                  <option value="Math">Math</option>
                  <option value="Science">Science</option>
                  <option value="Language Arts">Language Arts</option>
                  <option value="Social Studies">Social Studies</option>
                  <option value="Arts">Arts</option>
                  <option value="Music">Music</option>
                  <option value="Physical Education">Physical Education</option>
                </select>
              </div>

              <div className="form-group">
                <label htmlFor="grade_levels">Grade Levels</label>
                <input
                  type="text"
                  id="grade_levels"
                  value={formData.grade_levels}
                  onChange={(e) => setFormData({ ...formData, grade_levels: e.target.value })}
                  placeholder="e.g., K-2, 3-5, 6-8"
                />
              </div>
            </div>

            <div className="form-group">
              <label htmlFor="nys_standards">NYS Learning Standards (comma-separated)</label>
              <input
                type="text"
                id="nys_standards"
                value={formData.nys_standards}
                onChange={(e) => setFormData({ ...formData, nys_standards: e.target.value })}
                placeholder="e.g., CCSS.Math.K.CC.A.1, NYS.Science.MS.PS1"
              />
            </div>
          </>
        )}

        <div className="form-group">
          <label htmlFor="file">File *</label>
          <input
            ref={fileInputRef}
            type="file"
            id="file"
            required
            onChange={handleFileSelect}
            accept={uploadType === 'game' ? '.html,.htm' : '*'}
          />
          {file && (
            <div className="file-info">
              Selected: {file.name} ({(file.size / 1024 / 1024).toFixed(2)} MB)
            </div>
          )}
        </div>

        <div className="form-group">
          <label className="checkbox-label">
            <input
              type="checkbox"
              checked={formData.is_public}
              onChange={(e) => setFormData({ ...formData, is_public: e.target.checked })}
            />
            Make this publicly available (free for all)
          </label>
        </div>

        {uploadType === 'game' && !formData.is_public && (
          <div className="form-group">
            <label className="checkbox-label">
              <input
                type="checkbox"
                checked={formData.requires_enrollment}
                onChange={(e) => setFormData({ ...formData, requires_enrollment: e.target.checked })}
              />
              Require class enrollment to access
            </label>
          </div>
        )}

        {uploading && (
          <div className="progress-bar">
            <div className="progress-fill" style={{ width: `${uploadProgress}%` }} />
            <span className="progress-text">{uploadProgress.toFixed(0)}%</span>
          </div>
        )}

        {message && (
          <div className={`message ${message.type}`}>
            {message.text}
          </div>
        )}

        <button type="submit" className="submit-btn" disabled={uploading}>
          {uploading ? 'Uploading...' : `Upload ${uploadType === 'game' ? 'Game' : 'Material'}`}
        </button>
      </form>

      <style jsx>{`
        .instructor-upload-container {
          max-width: 800px;
          margin: 0 auto;
          padding: 24px;
          background: #ffffff;
          border-radius: 12px;
          box-shadow: 0 2px 8px rgba(0,0,0,0.1);
        }

        .upload-header {
          margin-bottom: 24px;
        }

        .upload-header h2 {
          margin: 0 0 8px 0;
          color: #2F6B65;
        }

        .upload-header p {
          margin: 0;
          color: #666;
        }

        .upload-type-selector {
          display: flex;
          gap: 12px;
          margin-bottom: 24px;
        }

        .type-btn {
          flex: 1;
          padding: 16px;
          border: 2px solid #e0e0e0;
          border-radius: 8px;
          background: #fff;
          font-size: 16px;
          font-weight: 600;
          cursor: pointer;
          transition: all 0.2s;
        }

        .type-btn:hover {
          border-color: #2F6B65;
        }

        .type-btn.active {
          border-color: #2F6B65;
          background: #2F6B65;

        }

          display: flex;
          flex-direction: column;
          gap: 16px;
        }

        .form-row {
          display: grid;
          grid-template-columns: 1fr 1fr;
          gap: 16px;
        }

        .form-group {
          display: flex;
          flex-direction: column;
          gap: 8px;
        }

        .form-group label {
          font-weight: 600;
          color: #333;
        }

        .form-group input[type="text"],
        .form-group input[type="file"],
        .form-group textarea,
        .form-group select {
          padding: 12px;
          border: 2px solid #e0e0e0;
          border-radius: 8px;
          font-size: 14px;
          font-family: inherit;
        }

        .form-group input:focus,
        .form-group textarea:focus,
        .form-group select:focus {
          outline: none;
          border-color: #2F6B65;
        }

        .checkbox-label {
          display: flex;
          align-items: center;
          gap: 8px;
          cursor: pointer;
        }

        .checkbox-label input[type="checkbox"] {
          width: 20px;

        }

        .file-info {
          padding: 8px 12px;

        .progress-fill {

    </div>
  );
}
```

## Storage Bucket Setup

Run this in Supabase SQL Editor:

```sql
-- Create storage bucket for instructor uploads
INSERT INTO storage.buckets (id, name, public)
VALUES ('instructor-uploads', 'instructor-uploads', true);

-- Enable RLS
ALTER TABLE storage.objects ENABLE ROW LEVEL SECURITY;

-- Policy: Instructors can upload to their own folder
CREATE POLICY "Instructors can upload files"
ON storage.objects FOR INSERT
WITH CHECK (
  bucket_id = 'instructor-uploads' AND
  (storage.foldername(name))[1] IN (
    SELECT id::text FROM profiles
    WHERE user_id = auth.uid() AND role = 'instructor'
  )
);

-- Policy: Anyone can read public files
CREATE POLICY "Public files are viewable by everyone"
ON storage.objects FOR SELECT
USING (bucket_id = 'instructor-uploads');

-- Policy: Instructors can update their own files
CREATE POLICY "Instructors can update own files"
ON storage.objects FOR UPDATE
USING (
  bucket_id = 'instructor-uploads' AND
  (storage.foldername(name))[1] IN (
    SELECT id::text FROM profiles
    WHERE user_id = auth.uid() AND role = 'instructor'
  )
);

-- Policy: Instructors can delete their own files
CREATE POLICY "Instructors can delete own files"
ON storage.objects FOR DELETE
USING (
  bucket_id = 'instructor-uploads' AND
  (storage.foldername(name))[1] IN (
    SELECT id::text FROM profiles
    WHERE user_id = auth.uid() AND role = 'instructor'
  )
);
```

## Features

- **Dual Upload Modes**: Games (HTML) and Materials (any file type)
- **Auto File Type Detection**: Intelligently detects file type from extension
- **Progress Tracking**: Real-time upload progress bar
- **Supabase Storage Integration**: Secure file storage with RLS
- **Public/Private Control**: Instructors choose access level
- **NYS Standards Tagging**: Link content to learning standards
- **Class Association**: Optionally link uploads to specific classes
- **Form Validation**: Required fields and file validation
- **Responsive Design**: Works on desktop and mobile

## Usage

```tsx
import InstructorUpload from '@/components/InstructorUpload';

<InstructorUpload
  instructorId="instructor-uuid-here"
  classId="optional-class-uuid"
/>
```
